ls : lista
cd : entra nas pastas 
mkdir : criar pastas 
ls -a : mostra arquivos ocultos 

cd .. volta um nivel
git add nomearquivo pastadedestino/

git status

git push origin master move os arquivos para o github

git pull origin master puxa o conteudo do git hub
